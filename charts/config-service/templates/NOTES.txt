config-service has been deployed.

Get the service URL:
{{- if eq .Values.service.type "ClusterIP" }}
  kubectl port-forward svc/{{ include "config-service.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
{{- end }}

Endpoints:
  GET  /config/:app     - resolved config for a single app
  GET  /config           - all configs merged
  GET  /config?apps=a,b  - selective configs
  GET  /health           - health + validation status
  POST /config           - reload configs from disk

Verify:
  curl http://localhost:{{ .Values.service.port }}/health
  curl http://localhost:{{ .Values.service.port }}/config/webapp
  curl http://localhost:{{ .Values.service.port }}/config/api
